class ClubDescriptionID(models.Model):
    CLUB_PALM_GRIP_CHOICES = (('LH','Left hand'),('RH','Right hand'))
    CLUB_TYPE_CHOICE = (
        ('DR','Driver'),('W','Wedge'),('P','Putter')
    )
    IRON_TYPE_CHOICE = (
        ('3i','3 iron'),('4i','4 iron'),('5i','5 iron'),('6i','6 iron'),
        ('7i','7 iron'),('8i','8 iron')
    )
    WOOD_TYPE_CHOICE = (
        ('3w','3 wood'),('2w','2 wood'),('1w','1 wood')
    )
    CLUB_SIZE_CHOICES = (('L','Ladies'),('M','Mens'),('J','Juniors'))
    clubName = models.CharField(max_length=50)
    clubPrice = models.FloatField(max_length=6)
    clubBrand = models.CharField(max_length=50)
    clubChoice = models.CharField(max_length=2, choices=CLUB_TYPE_CHOICE)
    ironChoice = models.CharField(max_length=2, choices=IRON_TYPE_CHOICE)
    woodChoice = models.CharField(max_length=2, choices=WOOD_TYPE_CHOICE)
    #clubImage = models.ImageField(blank=True, upload_to='images')
    clubSizeChoice = models.CharField(max_length=2, choices=CLUB_SIZE_CHOICES)
    gripChoice = models.CharField(max_length=2, choices=CLUB_PALM_GRIP_CHOICES)
    size = models.ManyToManyField('SizeID', blank=True)

 

class ClubSetsID(models.Model):
    CLUB_SET_SIZE = (('F','Full'),('H','Half'))
    CLUB_SIZE_CHOICES = (('L','Ladies'),('M','Mens'),('J','Juniors'))
    CLUB_PALM_GRIP_CHOICES = (('LH','Left hand'),('RH','Right hand'))
    setName = models.CharField(max_length=50)
    setPrice = models.FloatField(max_length=7)
    clubSetSize = models.CharField(max_length=1, choices=CLUB_SET_SIZE)
    clubSizeChoice = models.CharField(max_length=2, choices=CLUB_SIZE_CHOICES)
    gripChoice = models.CharField(max_length=2, choices=CLUB_PALM_GRIP_CHOICES)

class AccessoriesID(models.Model):
    accessName = models.CharField(max_length=50)
    accessPrice = models.FloatField(max_length=6)

class BallsID(models.Model):
    BALL_TYPE = (('La','Ladies'),('Pl','Play'),('Pr','Practice'))
    ballType = models.CharField(max_length=2,choices=BALL_TYPE)
    ballPrice = models.FloatField(max_length=6)
    ballBrand = models.CharField(max_length=50)

class GlovesID(models.Model):
    gloveGrip = models.CharField(max_length=50)
    gloveBrand = models.CharField(max_length=50)
    glovePrice = models.FloatField(max_length=6)
